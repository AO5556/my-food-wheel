    <script>
        const foods = ["小笼包", "生煎馒头", "葱油拌面", "排骨年糕", "麻辣烫", "黄焖鸡米饭", "炸鸡汉堡", "轻食沙拉", "螺蛳粉", "烧烤", "奶茶炸串", "日料寿司", "新疆炒米粉", "本帮红烧肉"];
        const colors = ["#f1c40f", "#e67e22", "#e74c3c", "#9b59b6", "#3498db", "#1abc9c", "#2ecc71", "#d35400", "#c0392b", "#8e44ad", "#2980b9", "#16a085", "#27ae60", "#f39c12"];
        
        const canvas = document.getElementById('wheelCanvas');
        const ctx = canvas.getContext('2d');
        const btn = document.getElementById('spin-button');
        const resultDisplay = document.getElementById('result');
        
        const numSegments = foods.length;
        const segmentAngle = 2 * Math.PI / numSegments;
        let currentRotation = 0;

        // 绘制转盘
        function drawWheel() {
            foods.forEach((food, i) => {
                const angle = i * segmentAngle;
                ctx.beginPath();
                ctx.fillStyle = colors[i % colors.length];
                ctx.moveTo(200, 200);
                ctx.arc(200, 200, 200, angle, angle + segmentAngle);
                ctx.lineTo(200, 200);
                ctx.fill();

                // 绘制文字
                ctx.save();
                ctx.translate(200, 200);
                ctx.rotate(angle + segmentAngle / 2);
                ctx.textAlign = "right";
                ctx.fillStyle = "white";
                ctx.font = "bold 16px Arial";
                ctx.fillText(food, 180, 5);
                ctx.restore();
            });
        }

        drawWheel();

        // 旋转功能
        btn.addEventListener('click', () => {
            resultDisplay.innerText = "正在挑选...";
            const randomSpin = Math.floor(Math.random() * 3600) + 1440; // 旋转 4-14 圈
            currentRotation += randomSpin;
            canvas.style.transform = `rotate(${currentRotation}deg)`;

            // 计算结果
            setTimeout(() => {
                const finalAngle = (360 - (currentRotation % 360)) % 360;
                const index = Math.floor((finalAngle / 360) * numSegments);
                resultDisplay.innerText = `今天吃：${foods[index]}`;
            }, 4000);
        });
    </script>
